Adios

Basic call

$.fn.adios({
	title : 'Hello, world.',
	buttons : [
		{
			callback : $.fn.adios.close,
			title : 'Oh hi there.'
		}
	]
});

All options

$.fn.adios({
	title : 'Notifications have been enabled',
	message : 'Notifications may include alerts, sounds and icon badges. These can be configured in Settings.',
	shade_callback : $.fn.adios.close,
	shade_transparent : true,
	buttons : [
		{
			callback : function(){
				alert('Cancelled!');
				$.fn.adios.close();
			},
			title : 'Cancel'
		},
		{
			callback : function(){
				alert('Okay!');
			},
			classname : 'jqueryadios_button_highlighted',
			href : '#new_page',
			title : 'OK'
		}
	]

});

Binding on click

Execute, then click to show notification

// <a href="#" class="adiosclick">Execute, then click to show notification</a>
$('a.adiosclick').adios({ 
	title : 'Notification title', 
	cancel_click : true,
	buttons : [
		{
			callback : $.fn.adios.close,
			title : 'OK'
		}
	]
});

Binding on click using data- attributes

Execute, then click to show notification 

// <a href="#" 
//   class="adiosdata" 
//   data-title="Notification title" 
//   data-buttons='[{"title" : "OK", "callback" : "$.fn.adios.close"}]'>
//     Execute, then click to show notification
// </a>
$('a.adiosdata').adios({cancel_click : true});
