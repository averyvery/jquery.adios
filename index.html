Adios

Basic call

$.fn.adios();

All options

$.fn.adios({
	title : 'Dental Appointment: 5pm',
	message : 'Notifications may include alerts, sounds and icon badges. These can be configured in Settings.',
	shade_callback : $.fn.adios.close,
	buttons : [
		{
			callback : function(){
				alert('Cancelled!');
				$.fn.adios.close();
			},
			title : 'Cancel'
		},
		{
			callback : function(){
				alert('Okay!');
			},
			classname : 'jqueryadios_button_highlighted',
			href : '#new_page',
			title : 'OK'
		}
	]

});

Binding on click

Click to show notification

$('a.adioslink').adios({ 
	title : 'Notification title', 
	cancel_click : true,
	message : '',
	buttons : [
		{
			callback : $.fn.adios.close,
			title : 'OK'
		}
	]
});
